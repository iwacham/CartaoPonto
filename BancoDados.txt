CREATE TABLE TBL_FUNCIONARIO(
	CPF VARCHAR2(11) PRIMARY KEY NOT NULL,
	DT_NASC TIMESTAMP NOT NULL,
	NOME VARCHAR2(80) NOT NULL,
	TELEFONE NUMBER(10) NOT NULL,
	CARGO VARCHAR2(50) NOT NULL,
	SENHA VARCHAR2(6) NOT NULL);
	
CREATE TABLE TBL_TIMECARD(
	CPF VARCHAR2(11) NOT NULL,
	DATA_PONTO VARCHAR2(10) NOT NULL,
	HORA_ENTRADA TIMESTAMP NOT NULL,
	HORA_SAIDA_INTERVALO TIMESTAMP,
	HORA_RETORNO_INTERVALO TIMESTAMP,
	HORA_SAIDA TIMESTAMP,
	DATA_FAZER_ALTERACAO VARCHAR2(10),
	HORA_FAZER_ALTERACAO VARCHAR2(10),
	OBSERVACAO VARCHAR2(100),
	STATUS NUMBER(1),
	CONSTRAINT FK_CPF FOREIGN KEY(CPF) REFERENCES TBL_FUNCIONARIO(CPF));
